// Binary Search in C with Sorted Array 
#include <stdio.h>
#include <stdlib.h>

// Function to print an array
void printArray(int arr[], int size)
{
    int i;
    for (i = 0; i < size; i++)
        printf("%d ", arr[i]);
    printf("\n");
}


// Binary Search using Recursive method
int binarySearch(int arr[], int data, int low, int high) {
  if (high >= low) {
    int mid = low + (high - low) / 2;

    // If found at mid, then return it
    if (arr[mid] == data)
      return mid;

    // Search the left half
    if (arr[mid] > data)
      return binarySearch(arr, data, low, mid - 1);

    // Search the right half
    return binarySearch(arr, data, mid + 1, high);
  }

  return -1;
}

int main(void) {
  int arr[] = {1,2,4,5,6, 7,9, 10, 15, 23};
  int arr_size = sizeof(arr) / sizeof(arr[0]);
  printf("Given array is \n");
  printArray(arr, arr_size);
    
  int data;
  printf("\nPlease enter the element to search for:");
  scanf("%d", &data);
  int result = binarySearch(arr, data, 0, arr_size - 1);
  if (result == -1)
    printf("Not found");
  else
    printf("Element is found at index %d", result);
  return 0;
}
